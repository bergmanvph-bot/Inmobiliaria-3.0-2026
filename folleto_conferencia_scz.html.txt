    <script>
        let currentSlide = 0;
        const totalSlides = document.querySelectorAll('.slide').length;
        const wrapper = document.getElementById('masterContainer');
        const dotNav = document.getElementById('dotNav');

        for (let i = 0; i < totalSlides; i++) {
            const dot = document.createElement('div');
            dot.className = `dot-item ${i === 0 ? 'active' : ''}`;
            dot.onclick = () => goToSlide(i);
            dotNav.appendChild(dot);
        }

        function updateUI() {
            wrapper.style.transform = `translateX(-${currentSlide * 100}vw)`;
            document.querySelectorAll('.dot-item').forEach((d, i) => { d.className = `dot-item ${i === currentSlide ? 'active' : ''}`; });
            const activeSlide = document.querySelectorAll('.slide')[currentSlide];
            const cat = activeSlide.getAttribute('data-cat');
            document.querySelectorAll('.category-menu span').forEach(s => s.classList.remove('active'));
            if (cat) {
                const label = document.getElementById(`nav-${cat}`);
                if (label) label.classList.add('active');
            }
        }

        function nextSlide() { if (currentSlide < totalSlides - 1) { currentSlide++; updateUI(); } }
        function prevSlide() { if (currentSlide > 0) { currentSlide--; updateUI(); } }
        function goToSlide(index) { currentSlide = index; updateUI(); }
        function goToSection(cat) {
            const firstOfCat = Array.from(document.querySelectorAll('.slide')).find(s => s.getAttribute('data-cat') === cat);
            if (firstOfCat) { currentSlide = parseInt(firstOfCat.getAttribute('data-index')); updateUI(); }
        }
        document.addEventListener('keydown', (e) => { if (e.key === 'ArrowRight' || e.key === ' ') nextSlide(); if (e.key === 'ArrowLeft') prevSlide(); });
        updateUI();
    </script>
